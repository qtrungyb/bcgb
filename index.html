// ========================================== // WORKFLOW DUYỆT BÁO CÁO // ========================================== window.trinhDuyetBaoCao = async function() { // Đã gỡ bỏ chốt chặn khóa form ở đây // Kiểm tra xem đã lưu chưa const key = getStorageKey(activeDate); let data = await LocalDB.getItem(key); if (!data || !data.isSaved) { return showToast("Vui lòng Lưu báo cáo trước khi Trình duyệt!", "error"); } promptPassword("Xác nhận TRÌNH DUYỆT báo cáo", async function(user) { if (user) { let btn = document.getElementById('btnTrinhDuyet'); if(btn) { btn.innerHTML = ` Đang gửi...`; btn.disabled = true; } try { currentReportSubmitted = true; currentReportApproved = false; currentRejectMessage = ""; await window.saveReportAll(true, true); logAction(`Đã TRÌNH DUYỆT báo cáo ngày ${activeDate}`); showToast("📤 Đã trình duyệt thành công! Báo cáo đang chờ Lãnh đạo duyệt.", "success"); } catch(e) { showToast("Lỗi khi trình duyệt: " + e.message, "error"); } finally { if(btn) { btn.innerHTML = ` Trình Duyệt`; btn.disabled = false; } } } }); } window.tuChoiBaoCao = async function() { promptPassword("Xác thực Lãnh đạo để TỪ CHỐI", async function(user) { if (user && ["admin", "quantri"].includes(user.role)) { openModal("Nhập lý do từ chối (bắt buộc):", async (lyDo) => { if (!lyDo || lyDo.trim() === "") { return showToast("Vui lòng nhập lý do từ chối để nhân viên sửa lại!", "error"); } let btn = document.getElementById('btnTuChoi'); if(btn) { btn.innerHTML = ` Đang xử lý...`; btn.disabled = true; } try { currentReportSubmitted = false; currentReportApproved = false; currentRejectMessage = lyDo.trim(); await window.saveReportAll(true, true); logAction(`Đã TỪ CHỐI báo cáo ngày ${activeDate}. Lý do: ${currentRejectMessage}`); showToast("❌ Đã từ chối báo cáo. Nhân viên có thể sửa lại.", "success"); } catch(e) { showToast("Lỗi khi từ chối: " + e.message, "error"); } finally { if(btn) { btn.innerHTML = ` Từ Chối`; btn.disabled = false; } } }); } else if (user) { showToast("Chỉ Lãnh đạo hoặc Admin mới có quyền Từ Chối!", "error"); } }); } window.duyetBaoCao = async function() { promptPassword("Xác thực Lãnh đạo để DUYỆT báo cáo", async function(user) { if (user && ["admin", "quantri"].includes(user.role)) { let btn = document.getElementById('btnDuyet'); if(btn) { btn.innerHTML = ` Đang duyệt...`; btn.disabled = true; } try { currentReportApproved = true; currentReportSubmitted = true; currentRejectMessage = ""; currentApprovalInfo = { nguoiDuyet: user.fullname, thoiGianDuyet: new Date().toLocaleTimeString('vi-VN', {hour: '2-digit', minute:'2-digit'}) + ' ' + new Date().toLocaleDateString('vi-VN') }; await window.saveReportAll(true, true); logAction(`Đã DUYỆT báo cáo ngày ${activeDate}`); showToast("✅ Đã phê duyệt báo cáo thành công!", "success"); } catch(e) { showToast("Lỗi khi duyệt: " + e.message, "error"); } finally { if(btn) { btn.innerHTML = ` Duyệt Báo Cáo`; btn.disabled = false; } } } else if (user) { showToast("Chỉ Lãnh đạo hoặc Admin mới có quyền Duyệt báo cáo!", "error"); } }); } window.huyDuyetBaoCao = async function() { promptPassword("Xác thực Lãnh đạo để HỦY DUYỆT", async function(user) { if (user && ["admin", "quantri"].includes(user.role)) { const isConfirmed = await openCancelApprovalConfirm(); if (!isConfirmed) return; try { let btn = document.getElementById('btnHuyDuyet'); if(btn) { btn.innerHTML = ` Đang xử lý...`; btn.disabled = true; } currentReportApproved = false; currentReportSubmitted = true; currentRejectMessage = ""; currentApprovalInfo = {}; await window.saveReportAll(true, true); logAction(`Đã HỦY DUYỆT báo cáo ngày ${activeDate}`); showToast("🔓 Đã hủy phê duyệt. Báo cáo chuyển về trạng thái Chờ Duyệt!", "success"); } catch(e) { showToast("Lỗi khi hủy duyệt: " + e.message, "error"); } finally { let btn = document.getElementById('btnHuyDuyet'); if(btn) { btn.innerHTML = ` Hủy Phê Duyệt`; btn.disabled = false; } } } else if (user) { showToast("Chỉ Lãnh đạo hoặc Admin mới có quyền Hủy Duyệt báo cáo!", "error"); } }); } // Initialize application check checkInitialLogin();
